    ;

containerShipParamTable OBJECT-TYPE
    SYNTAX      SEQUENCE OF containerShipParamEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
	"grupo com objetos que representam um container novo a ser criado"
    ::= { containerSHIPMIB 1 }


containerShipParamEntry OBJECT-TYPE
    SYNTAX      containerShipParamTableEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
	"Uma linha da tabela com 3 objetos: índice da imagem, nome e flag"
    INDEX   { indImage }
    ::= { containerShipParamTable 1 }

containerShipParamTableEntry ::= SEQUENCE {
	indImage	Integer32,
	flag	Integer32,
	nome	DisplayString
}


containerShipImageTable OBJECT-TYPE
    SYNTAX      SEQUENCE OF containerShipImageEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
	"grupo com a tabela de imagens disponíveis"
    ::= { containerSHIPMIB 2 }

containerShipImageEntry OBJECT-TYPE
    SYNTAX      containerShipImageTableEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
	"Uma linha da tabela pelo menos 2 objetos: INDEX e image"
    INDEX   { index }
    ::= { containerShipImageTable 1 }

containerShipImageTableEntry ::= SEQUENCE {
	index	Integer32,
	image	DisplayString
}

containerShipContainerTable OBJECT-TYPE
    SYNTAX      SEQUENCE OF containerShipContainerEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
	"grupo com a tabela de containers que foramcriados"
    ::= { containerSHIPMIB 3 }

containerShipContainerEntry OBJECT-TYPE
    SYNTAX      containerShipContainerTableEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
	"Uma linha da tabela pelo menos 4 objetos: INDEX, Name,indice da image, Status (up, down,
removing, creating e changing)e Procesor (percentagem de utilização desse container)"
    INDEX   { index }
    ::= { containerShipContainerTable 1 }

containerShipContainerTableEntry ::= SEQUENCE {
	index	Integer32,
    Name DisplayString,
	image DisplayString,
    Status DisplayString, 
    Procesor DisplayString
}
