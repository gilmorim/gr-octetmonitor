containerShipParamTable OBJECT-TYPE
    SYNTAX      SEQUENCE OF ContainerShipParamTableEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
	"grupo com objetos que representam um container novo a ser criado"
    ::= { containerSHIPMIB 1 }


containerShipParamEntry OBJECT-TYPE
    SYNTAX      ContainerShipParamTableEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
	"Uma linha da tabela com 3 objetos indice da imagem, nome e flag"
    INDEX   { indexp }
    ::= { containerShipParamTable 1 }

ContainerShipParamTableEntry ::= SEQUENCE {
    indexp Integer32,
	indImagep Integer32,
	flagp Integer32,
	nomep DisplayString
}

    SYNTAX      Integer32 (1..32)
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
	"indice que indentifica a linha da tabela "
    ::= { containerShipParamEntry 1 }
indImagep OBJECT-TYPE
    SYNTAX      Integer32 (1..32)
    MAX-ACCESS  read-write
    STATUS      current
    DESCRIPTION
	"indice da imagem do container novo a ser criado "
    ::= { containerShipParamEntry 2 }

flagp OBJECT-TYPE
    SYNTAX      Integer32 (1..32)
    MAX-ACCESS  read-write
    STATUS      current
    DESCRIPTION
	"flag do container novo a ser criado. Indica se devera ser criado o container"
    ::= { containerShipParamEntry 3 }

nomep OBJECT-TYPE
    SYNTAX      DisplayString
    MAX-ACCESS  read-write
    STATUS      current
    DESCRIPTION
	"nome do container novo a ser criado"
    ::= { containerShipParamEntry 4 }

    SYNTAX SEQUENCE OF ContainerShipStatusTableEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
	"Informacao estistica da utilizacao da mib"
    ::= { containerSHIPMIB 4 }

containerShipStatusEntry OBJECT-TYPE
    SYNTAX      ContainerShipStatusTableEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
	"Uma linha da tabela contem varias Informacoes estisticas sobre a utilizacao: id do utilizador, index, tempo de inicio e tempo final da utilizacao"

    ::= { containerShipStatusEntry 1 }

    STATUS      current
    DESCRIPTION

    ::= { containerShipStatusEntry 2 }

timeBegins OBJECT-TYPE
    SYNTAX      TimeTicks
    MAX-ACCESS  read-create
    STATUS      current
    DESCRIPTION
	"etiqueta que indica o inicio da criacao do container"
    ::= { containerShipStatusEntry 3 }

timeFinals OBJECT-TYPE
    SYNTAX      TimeTicks
    MAX-ACCESS  read-create
    STATUS      current
    DESCRIPTION
	"etiqueta que indica o final do uso do container"
